\section{Sobre análise de formas}

\frame{
  \frametitle{Análise de formas}
  
  \begin{itemize}
  \item Formas: palavras na linguagem visual
  \item Análise de modelos em 3D tem custo computacional muito alto:
    abordagem utilizando formas em 2D é viável
  \item Caráter multidisciplinar: medicina, biologia, física, etc.
  \end{itemize}
}

\subsection{Etapas}

\frame{
  \frametitle{Etapas envolvidas}

  \begin{itemize}
  \item Aquisição
  \item Detecção de formas
  \item Filtro de ruído
  \item Operação sobre formas: normalização
  \item Transformação: evolução, representação
  \item Classificação: supervisionada, não-supervisionada,
    similaridade, \emph{matching}
  \end{itemize} 
}

\section{Relação entre pesquisa e análise de formas}

\frame{
  \frametitle{Relação entre pesquisa e análise de formas}

  \begin{itemize}
  \item Radioterapia: \emph{tracking} do marcador, registro de imagens
  \item \emph{Deep Learning}: diferentes representações e
    características podem ajudar a gerar visualizações das camadas
  \end{itemize}
}
